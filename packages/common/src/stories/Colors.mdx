import { Meta, Story } from "@storybook/blocks";

import { colors } from "~/styles/colors";
import { cn } from "utils";

export function StoriesColor(props) {
  const { title, value, className } = props;
  return (
    <div
      className="flex flex-col items-center justify-center text-xs"
      style={{ width: "80px", gap: "4px" }}
    >
      <div
        className={className}
        style={{
          width: "80px",
          height: "50px",
          borderRadius: "5px",
          backgroundColor: value,
        }}
      />
      <div
        className="text-foreground"
        style={{ whiteSpace: "nowrap", textAlign: "center" }}
      >
        {title}
      </div>
      <div className="text-grey-400">{value}</div>
    </div>
  );
}

export function renderPalette(name) {
  return Object.entries(colors[name]).map(([key, value]) => (
    <StoriesColor key={key} title={`${name}-${key}`} value={value} />
  ));
}
