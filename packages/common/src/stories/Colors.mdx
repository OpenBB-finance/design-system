import { Meta, Story } from "@storybook/blocks";

import { colors, gradients, radials } from "common/src/styles/colors";
import { cn } from "utils";

export function StoriesColor(props) {
  const { title, color, image, className } = props;
  return (
    <div
      className="flex flex-col items-center justify-center text-xs text-grey-100 dark:text-black"
      style={{ width: "80px", gap: "4px" }}
    >
      <div
        className={className}
        style={{
          width: "80px",
          height: "50px",
          borderRadius: "5px",
          backgroundColor: color,
          backgroundImage: image,
        }}
      />
      <div style={{ whiteSpace: "nowrap", textAlign: "center" }}>{title}</div>
      <div className="text-grey-400">{color}</div>
    </div>
  );
}

export function renderPalette(name) {
  return Object.entries(colors[name]).map(([key, value]) => (
    <StoriesColor key={key} title={`${name}-${key}`} color={value} />
  ));
}

export function renderGradients() {
  return Object.entries(gradients).map(([key, value]) => (
    <StoriesColor key={key} title={`${key}`} image={value} />
  ));
}

export function renderRadials() {
  return Object.entries(radials).map(([key, value]) => (
    <StoriesColor
      key={key}
      title={`${key}`}
      image={value}
      className="!aspect-square !h-auto"
    />
  ));
}

<Meta
  title="Colors"
  parameters={{
    themes: {
      themeOverride: "dark",
    },
  }}
/>
